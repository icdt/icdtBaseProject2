checklist-model

注意事項: vm.allOptions 與 vm.theOptions 陣列中的元素須是相同類型or物件

UI

  • item.id: {{item.id}}, item.text: {{item.text}}

html


        <label class="checkbox-inline" ng-repeat="item in vm.allOptions">
            <input type="checkbox" checklist-model="vm.theOptions" checklist-value="item" />{{item.text}}
        </label>

app.js


            app = angular.module('app', ['checklist-model']);

controller.js


        app.controller('ChecklistModelCtrl', ['$scope', function ($scope) {
            // 宣告vm
            $scope.vm = {};
            
            // 所有的選項
            $scope.vm.allOptions = [
                {
                    id: 1,
                    text: 'apple'
                },
                {
                    id: 2,
                    text: 'banana'
                },
                {
                    id: 3,
                    text: 'car'
                },
                {
                    id: 4,
                    text: 'duck'
                }
            ];
            
            // 使用者的選擇
            $scope.vm.theOptions = [];
        }]);
